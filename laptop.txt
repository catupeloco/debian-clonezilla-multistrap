# Cleaning script and resetting terminal for multiple run fixes
reset
printf "\033c"
echo --Installing git
sudo apt install git -y >/dev/null 2>&1

if [ -d $HOME/debian-clonezilla-multistrap ] ; then 
	rm -rf $HOME/debian-clonezilla-multistrap
fi

echo --Cloning repository
while [ ! -f debian-clonezilla-multistrap/setup.sh ] ; do
	git clone https://github.com/catupeloco/debian-clonezilla-multistrap >/dev/null 2>&1
done

echo --Running main script
time sudo bash $HOME/debian-clonezilla-multistrap/setup.sh
